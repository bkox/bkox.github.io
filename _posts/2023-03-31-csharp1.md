title:  "DivideByZeroException 에 관한 예제"
categories: "csharp"
---

```csharp
int number1 = 3000;
int number2 = 0;
try
{
	Console.WriteLine(number1 / number2);
}
catch (DivideByZeroException e)
{
	Console.WriteLine(e.Message);
}

var a = 10.0 / 0.0;

Console.WriteLine("10.0 / 0.0 = " + a);
```
